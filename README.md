# MySpinnerView
Android自定义Spinner下拉选择器，由popupwindow+recyclerview实现

```
//spinner_view右侧箭头颜色    
app:spinner_arrow_color="@color/black"
//spinner_view右侧箭头和文字的距离    
app:spinner_arrow_padding="5dp"
//item为空时默认显示文字    
app:spinner_hint_text="请选择"
//spinner_view文字颜色    
app:spinner_text_color="@color/black"
//spinner_view文字大小默认sp    
app:spinner_text_size="15"
//spinner_view的内间距 
app:spinner_padding_left="6dp"
app:spinner_padding_right="6dp"
app:spinner_padding_top="3dp"
app:spinner_padding_bottom="3dp"
//点击弹出来popup的上下内间距
app:popup_padding_top="5dp"
app:popup_padding_bottom="5dp"
//是否显示item选中是的对钩图片
app:item_show_select_image="true"
//item选中时的颜色，文字和图片颜色一致
app:item_select_color="#FF0000"
//item中文字大小，默认sp
app:item_text_size="15"
//每个item的内边距
app:item_padding_left="12dp"
app:item_padding_right="12dp"
app:item_padding_top="3dp"
app:item_padding_bottom="3dp"
```
